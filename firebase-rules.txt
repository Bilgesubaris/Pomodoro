// Firebase Firestore Güvenlik Kuralları
// Bu kuralları Firebase Console > Firestore Database > Rules kısmına yapıştır

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Kullanıcılar kendi verilerine erişebilir
    match /users/{userId} {
      // Kullanıcı kendi profil bilgilerine erişebilir
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Kullanıcının sessions alt koleksiyonuna erişim
      match /sessions/{sessionId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
  }
}
